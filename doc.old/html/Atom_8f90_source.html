212  function isconnectedwith( this, other, alpha ) result( output )
213  class(atom), intent(in) :: this
214  class(atom), intent(in) :: other
215  real(8), optional :: alpha
216  logical :: output
217 
218  real(8) :: dist, cutoff
219  real(8) :: effalpha
220

222  if( present(alpha) ) effalpha = alpha
223 
224  dist = sqrt(sum((this.r - other.r)**2))
225  cutoff = atomicelementsdb_instance.covalentradius( this.symbol ) + atomicelementsdb_instance.covalentradius( other.symbol )
