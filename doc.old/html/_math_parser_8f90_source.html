175  subroutine parsefunction( this, funcStr, var )
176  class(mathparser) :: this
177  character(*), intent(in) :: funcstr
178  character(*), intent(in) :: var(:)
179 
180  character(len(funcStr)) :: func ! Function string, local use
181 
182  allocate( ipos(len_trim(funcstr)) ) ! Char. positions in orig. string
183  func = funcstr ! Local copy of function string
184  call replace( '**', '^ ', func ) ! Exponent into 1-Char. format
185  call removespaces( func ) ! Condense function string
186  call checksyntax( this, func, funcstr, var )
187  deallocate( ipos )
188 
189  call compile( this, func, var ) ! Compile into bytecode
190  end subroutine parsefunction

197  function evaluatefunction( this, val ) result( res )
198  class(mathparser) :: this
199  real(8), intent(in) :: val(:)
200  real(8) :: res
201  integer :: ip ! Instruction pointer
202  integer :: dp ! Data pointer
203  integer :: sp ! Stack pointer
204 
205  dp = 1
206  sp = 0
207  do ip=1,this.bytecodesize
208  select case( this.bytecode(ip) )
209  case( c_immed )
210  sp=sp+1
211  this.stack(sp) = this.immed(dp)
212  dp = dp+1
213  case( c_neg )
214  this.stack(sp) = -this.stack(sp)
215  case( c_add )
216  this.stack(sp-1) = this.stack(sp-1)+this.stack(sp)
217  sp = sp-1
218  case( c_sub )
219  this.stack(sp-1) = this.stack(sp-1)-this.stack(sp)
220  sp = sp-1
221  case( c_mul )
222  this.stack(sp-1) = this.stack(sp-1)*this.stack(sp)
223  sp = sp-1
224  case( c_div )
225  if( this.stack(sp) == zero ) then
226  this.error = 1
227  res = zero
228  return
229  endif
230  this.stack(sp-1) = this.stack(sp-1)/this.stack(sp)
231  sp = sp-1
232  case( c_pow )
233  this.stack(sp-1) = this.stack(sp-1)**this.stack(sp)
234  sp = sp-1
235  case( c_abs )
236  this.stack(sp) = abs( this.stack(sp) )
237  case( c_exp )
238  this.stack(sp) = exp( this.stack(sp) )
239  case ( c_log10 )
240  if( this.stack(sp) <= zero ) then
241  this.error = 3
242  res = zero
243  return
244  endif
245  this.stack(sp) = log10( this.stack(sp) )
246  case( c_log )
247  if( this.stack(sp) <= zero ) then
248  this.error = 3
249  res = zero
250  return
251  endif
252  this.stack(sp) = log( this.stack(sp) )
253  case( c_sqrt )
254  if( this.stack(sp) < zero ) then
255  this.error = 3
256  res = zero
257  return
258  endif
259  this.stack(sp) = sqrt( this.stack(sp) )
260  case( c_sinh )
261  this.stack(sp) = sinh( this.stack(sp) )
262  case( c_cosh )
263  this.stack(sp) = cosh( this.stack(sp) )
264  case( c_tanh )
265  this.stack(sp) = tanh( this.stack(sp) )
266  case( c_sin )
267  this.stack(sp) = sin( this.stack(sp) )
268  case( c_cos )
269  this.stack(sp) = cos( this.stack(sp) )
270  case( c_tan )
271  this.stack(sp) = tan( this.stack(sp) )
272  case( c_asin )
273  if( (this.stack(sp)<-1.0_8) .or. (this.stack(sp)>1.0_8) ) then
274  this.error = 4
275  res = zero
276  return
277  endif
278  this.stack(sp) = asin( this.stack(sp) )
279  case( c_acos )
280  if( (this.stack(sp)<-1.0_8) .or. (this.stack(sp)>1.0_8) ) then
281  this.error = 4
282  res = zero
283  return
284  endif
285  this.stack(sp) = acos( this.stack(sp) )
286  case( c_atan )
287  this.stack(sp) = atan( this.stack(sp) )
288  case( c_sgn )
289  this.stack(sp) = sign( 1.0_8, this.stack(sp) )
290 #define ustep(x) 0.5_8*( 1.0_8 + tanh( 1.0d12*(x) ) )
291  case( c_ustep )
292  this.stack(sp) = ustep( this.stack(sp) )
293  case( c_ubox )
294  this.stack(sp) = ustep( this.stack(sp)+0.5_8 ) - ustep( this.stack(sp)-0.5_8 )
295 #undef ustep
296  case default
297  sp = sp+1
298  this.stack(sp) = val( this.bytecode(ip)-c_varbegin+1 )
299  end select
300  end do
301 
302  this.error = 0
303  res = this.stack(1)
304  end function evaluatefunction

313  subroutine checksyntax( this, func, funcStr, var )
314  class(mathparser) :: this
315  character(*), intent(in) :: func
316  character(*), intent(in) :: funcstr
317  character(*), dimension(:), intent(in) :: var
318 
319  integer :: n
320  character(1) :: c
321  real(8) :: r
322  logical :: err
323  integer :: parcounter ! Parenthesis counter
324  integer :: j,ib,in,lfunc
325 
326  j = 1
327  parcounter = 0
328  lfunc = len_trim(func)
329 
330  do while( .true. )
331  if( j > lfunc) call parseerrmsg( j, funcstr )
332  c = func(j:j)
333 
334  ! Check for valid operand (must appear)
335  if( c == '-' .OR. c == '+' ) then ! Check for leading - or +
336  j = j+1
337  if( j > lfunc) then
338  call parseerrmsg(j, funcstr, 'Missing operand' )
339  end if
340  c = func(j:j)
341  if( any(c == str_operators)) call parseerrmsg( j, funcstr, 'Multiple operators' )
342  end if
343  n = mathfunctionindex(func(j:))
344  if( n > 0) then ! Check for math function
345  j = j+len_trim(str_functions(n))
346  if( j > lfunc) call parseerrmsg( j, funcstr, 'Missing function argument' )
347  c = func(j:j)
348  if( c /= '(' ) call parseerrmsg( j, funcstr, 'Missing opening parenthesis' )
349  end if
350  if( c == '(' ) then ! Check for opening parenthesis
351  parcounter = parcounter+1
352  j = j+1
353  cycle
354  end if
355  if( scan(c,'0123456789.' ) > 0 ) then ! Check for number
356  r = strtoreal(func(j:),ib,in,err)
357  if( err) call parseerrmsg( j, funcstr, 'Invalid number format: '//func(j+ib-1:j+in-2))
358  j = j+in-1
359  if( j > lfunc) exit
360  c = func(j:j)
361  else ! Check for variable
362  n = variableindex(func(j:),var,ib,in)
363  if( n == 0) call parseerrmsg( j, funcstr, 'Invalid element: '//func(j+ib-1:j+in-2))
364  j = j+in-1
365  if( j > lfunc) exit
366  c = func(j:j)
367  end if
368  do while (c == ')' ) ! Check for closing parenthesis
369  parcounter = parcounter-1
370  if( parcounter < 0) call parseerrmsg( j, funcstr, 'Mismatched parenthesis' )
371  if( func(j-1:j-1) == '(' ) call parseerrmsg(j-1, funcstr, 'Empty parentheses' )
372  j = j+1
373  if( j > lfunc) exit
374  c = func(j:j)
375  end do
376 
377  ! Now, we have a legal operand: A legal operator or end of string must follow
378  if( j > lfunc) exit
379  if( any(c == str_operators)) then ! Check for multiple operators
380  if( j+1 > lfunc) call parseerrmsg( j, funcstr)
381  if( any(func(j+1:j+1) == str_operators)) call parseerrmsg(j+1, funcstr, 'Multiple operators' )
382  else ! Check for next operand
383  call parseerrmsg( j, funcstr, 'Missing operator' )
384  end if
385 
386  ! Now, we have an operand and an operator: the next loop will check for another
387  ! operand (must appear)
388  j = j+1
389  end do
390 
391  if( parcounter > 0 ) then
392  call parseerrmsg( j, funcstr, 'Missing )' )
393  end if
394  end subroutine checksyntax

608  recursive subroutine compilesubstr( this, f, b, e, var )
609  class(mathparser) :: this
610  character(*), intent(in) :: f
611  integer, intent(in) :: b,e
612  character(*), intent(in) :: var(:)
613 
614  integer :: n
615  integer :: b2,j,k,io
616  character(*), parameter :: calpha = 'abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ'
617 
618  ! Check for special cases of substring
619  if( f(b:b) == '+' ) then ! Case 1: f(b:e) = '+...'
620 ! write(*,*)'1. f(b:e) = "+..."'
621  call compilesubstr( this, f, b+1, e, var )
622  return
623  elseif( completelyenclosed( f, b, e ) ) then ! Case 2: f(b:e) = '(...)'
624 ! write(*,*)'2. f(b:e) = "(...)"'
625  call compilesubstr( this, f, b+1, e-1, var )
626  return
627  elseif( scan(f(b:b),calpha) > 0 ) then
628  n = mathfunctionindex( f(b:e) )
629  if( n > 0 ) then
630  b2 = b + index( f(b:e), '(' ) - 1
631  if( completelyenclosed( f, b2, e ) ) then ! Case 3: f(b:e) = 'fcn(...)'
632 ! write(*,*)'3. f(b:e) = "fcn(...)"'
633  call compilesubstr( this, f, b2+1, e-1, var )
634  call addcompiledbyte( this, n )
635  return
636  end if
637  end if
638  elseif( f(b:b) == '-' ) then
639  if( completelyenclosed(f, b+1, e) ) then ! Case 4: f(b:e) = '-(...)'
640 ! write(*,*)'4. f(b:e) = "-(...)"'
641  call compilesubstr( this, f, b+2, e-1, var )
642  call addcompiledbyte( this, c_neg )
643  return
644  elseif( scan( f(b+1:b+1), calpha ) > 0 ) then
645  n = mathfunctionindex( f(b+1:e) )
646  if( n > 0 ) then
647  b2 = b+index(f(b+1:e),'(')
648  if( completelyenclosed( f, b2, e ) ) then ! Case 5: f(b:e) = '-fcn(...)'
649 ! write(*,*)'5. f(b:e) = "-fcn(...)"'
650  call compilesubstr( this, f, b2+1, e-1, var)
651  call addcompiledbyte( this, n )
652  call addcompiledbyte( this, c_neg )
653  return
654  end if
655  end if
656  endif
657  end if
658 
659  ! Check for operator in substring: check only base level (k=0), exclude expr. in ()
660  do io=c_add,c_pow ! Increasing priority +-*/^
661  k = 0
662  do j=e,b,-1
663  if( f(j:j) == ')' ) then
664  k = k+1
665  elseif( f(j:j) == '(' ) then
666  k = k-1
667  end if
668  if( k == 0 .and. f(j:j) == str_operators(io) .and. isbinaryoperator(j, f) ) then
669  if( any(f(j:j) == str_operators(c_mul:c_pow)) .and. f(b:b) == '-' ) then ! Case 6: F(b:e) = '-...Op...' with Op > -
670 ! write(*,*)'6. f(b:e) = "-...Op..." with Op > -'
671  call compilesubstr( this, f, b+1, e, var )
672  call addcompiledbyte( this, c_neg )
673  return
674  else ! Case 7: f(b:e) = '...BinOp...'
675 ! write(*,*)'7. Binary operator',f(j:j)
676  call compilesubstr( this, f, b, j-1, var )
677  call compilesubstr( this, f, j+1, e, var )
678  call addcompiledbyte( this, operatorindex(str_operators(io)) )
679  this.stackptr = this.stackptr - 1
680  return
681  end if
682  end if
683  end do
684  end do
685 
686  ! check for remaining items, i.e. variables or explicit numbers
687  b2 = b
688  if( f(b:b) == '-' ) b2 = b2+1
689  n = mathitemindex( this, f(b2:e), var )
690 ! write(*,*)'8. AddCompiledByte ', n
691  call addcompiledbyte( this, n )
692  this.stackptr = this.stackptr + 1
693  if( this.stackptr > this.stacksize ) this.stacksize = this.stacksize + 1
694  if( b2 > b ) call addcompiledbyte( this, c_neg )
695  end subroutine compilesubstr

817  function strtoreal( str, ibegin, inext, error ) result ( res )
818  character(*), intent(in) :: str
819  integer, optional, intent(out) :: ibegin
820  integer, optional, intent(out) :: inext
821  logical, optional, intent(out) :: error
822  real(8) :: res

824  integer :: ib,in,istat

826  logical :: inman ! .T. in mantissa of number
827  logical :: pflag ! .T. after 1st '.' encountered
828  logical :: eflag ! .T. at exponent identifier 'eEdD'
829  logical :: inexp ! .T. in exponent of number
830  logical :: dinman ! .T. if at least 1 digit in mant.
831  logical :: dinexp ! .T. if at least 1 digit in exp.
832  logical :: err ! Local error flag
833 
834  bflag=.true.
835  inman=.false.
836  pflag=.false.
837  eflag=.false.
838  inexp=.false.
839  dinman=.false.
840  dinexp=.false.
841 
842  ib = 1
843  in = 1
844 
845  do while( in <= len_trim(str) )
846  select case( str(in:in) )
847  case( ' ' ) ! Only leading blanks permitted
848  ib = ib+1
849 
850  if( inman .or. eflag .or. inexp ) then
851  exit
852  end if
853  CASE( '+', '-' ) ! Permitted only
854  if( bflag ) then
855  inman=.true.; bflag=.false. ! at beginning of mantissa
856  else if( eflag ) then
857  inexp=.true.; eflag=.false. ! at beginning of exponent
858  else
859  exit ! - otherwise STOP
860  endif
861  case( '0' : '9' ) ! Mark
862  if( bflag ) then
863  inman=.true.; bflag=.false. ! beginning of mantissa
864  else if( eflag ) then
865  inexp=.true.; eflag=.false. ! beginning of exponent
866  endif
867 
868  if( inman ) dinman=.true. ! Mantissa contains digit
869  if( inexp ) dinexp=.true. ! Exponent contains digit
870  case( '.' )
871  if( bflag ) then
872  pflag=.true. ! mark 1st appearance of '.'
873  inman=.true.; bflag=.false. ! mark beginning of mantissa
874  else if( inman .and. .not. pflag ) then
875  pflag=.true. ! mark 1st appearance of '.'
876  else
877  exit ! otherwise STOP
878  end if
879  CASE( 'e', 'E', 'd', 'D' ) ! Permitted only
880  if( inman ) then
881  eflag=.true.; inman=.false. ! following mantissa
882  else
883  exit ! otherwise STOP
884  endif
885  case default
886  exit ! STOP at all other characters
887  end select
888  in = in+1
889  end do
890 
891  err = ( ib > in-1 ) .or. ( .not. dinman ) .or. ( (eflag .or. inexp) .and. .not. dinexp )
892 
893  if( err ) then
894  res = 0.0_8
895  else
896  read( str(ib:in-1), *, iostat=istat ) res
897  err = istat /= 0
898  end if
899 
900  if( present(ibegin) ) ibegin = ib
901  if( present(inext) ) inext = in
902  if( present(error) ) error = err
903  end function strtoreal
