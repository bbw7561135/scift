64  class(spline) :: this

72  real(8) :: t
73  real(8) :: yt
74  integer :: i
75 
76  if( present(tol) ) then
77  efftol = tol
78  else
79  efftol = 1.0e-12
80  end if
81 
82  call this.nfunc.copy( nfunc )
83  this.size = nfunc.npoints()
84 
85  allocate( _b(_n) )
86  allocate( a(_n) )
87  allocate( c(_n) )
88  allocate( s(_n) )
89 
90  _x = nfunc.xgrid.data
91  _y = nfunc.farray
92 
93  _b(1) = 0.0_8
94  _b(_n) = 0.0_8
95  a(1) = _x(2)-_x(1)
96  c(1) = (_y(2)-_y(1))/a(1)
97 
98  do i=2, _n-1
99  a(i) = _x(i+1)-_x(i)
100  c(i) = (_y(i+1)-_y(i))/a(i)
101  _b(i) = 2.0_8*(c(i)-c(i-1))/(a(i)+a(i-1))
102  s(i) = 1.5_8*_b(i)
103  end do
104 
105  do while ( t > efftol )
106  t = 0.0_8
107 
108  ! oblige derivee a etre horizontale pour extremitee de la courbe
109  ! pou supprimer enlever les deux cartes suivantes
110  _b(_n) = 3.d0*(_y(_n-1)-_y(_n))/((_x(_n)-_x(_n-1))**2)+_b(_n-1)*0.5d0
111  _b(_n) = _b(_n)-_b(_n-1)
112 
113  do i=2,_n-1
114  yt = 0.5_8*((_b(i+1)-_b(i-1))/(1.0_8+a(i)/a(i-1))-_b(i+1))-_b(i)+s(i)
115  yt = yt*4.0_8*(2.0_8-dsqrt(3.0_8))
116 
117  if( abs(yt) > t ) then
118  t = abs(yt)
119  end if
120 
121  _b(i)=yt+_b(i)
122  end do
123 
124  _b(_n) = _b(_n)-0.5_8*yt
125  end do
126 
127  deallocate( a )
128  deallocate( c )
129  deallocate( s )
130  end subroutine init

183  function evaluate( this, xValue ) result( output )
184  implicit none
185  class(spline) :: this
186  real(8), intent(in) :: xvalue
187  real(8) :: output
188 
189  integer :: i, k
190  real(8) :: xi
191  real(8) :: dx, dy, db
192  real(8) :: pa, pb, pc, pd
193 
194  i=0
195  do k=1,_n-1
196  xi = xvalue-_x(k)
197  dx = xi*( _x(k+1)-xvalue )
198 
199  if( dx >= 0.0_8 ) then
200  i=k
201  exit
202  end if
203  end do
204 
205  dy = _y(i+1)-_y(i)
206  dx = _x(i+1)-_x(i)
207  db = _b(i+1)-_b(i)
208  pa = db/(6.0_8*dx)
209  pb = (_b(i)+_b(i)+_b(i+1))/6.0_8
210  pc = dy/dx-dx*pb
211  pd = _y(i)
212  pb = pb-dx*pa
213 
214  output = pd+xi*(pc+xi*(pb+xi*pa))
215  end function evaluate
