91 
92  ! El peor de los casos es que todas las lineas sean datos
93  allocate( x(stream.numberoflines) )
94  allocate( y(stream.numberoflines) )
95 
96  ndata = 1
97  do while( .not. stream.eof() )
98  line = stream.readline( ccommentseff )
99 
100  if( len(line) /= 0 ) then
101  call buffer.fromfstring( line )
102  call buffer.split( tokens, " " )
103 
104  if( ndata == 1 .and. .not. ( size(tokens) >= size(columnseff) ) ) then
105  write(*,*) "### ERROR ### NFunction.fromFStream(): Number of columns in file ("&
106  //trim(fstring_frominteger(size(tokens)))&
107  //") are not in agree with the parameter columns ("&
108  //trim(fstring_frominteger(size(columnseff)))//"). nFile >= nCols"
109  stop
110  end if
111 
112  if( size(columnseff) == 3 ) then
113  if( columnseff(1) <= size(tokens) .and. columnseff(2) <= size(tokens) .and. columnseff(3) <= size(tokens) ) then
114  if( len_trim(tokens(columnseff(1))) /= 0 .and. len_trim(tokens(columnseff(2))) /= 0 .and. len_trim(tokens(columnseff(3))) /= 0 ) then
115  read( tokens(columnseff(1)),* ) x(ndata)
116 
117  buffer = "("//trim(tokens(columnseff(2)))//", "//trim(tokens(columnseff(3)))//")"
118  read( buffer.fstr,* ) y(ndata)
119  ndata = ndata + 1
120  end if
121  end if
122  else if( size(columnseff) == 2 ) then
123  if( columnseff(1) <= size(tokens) .and. columnseff(2) <= size(tokens) ) then
124  if( len_trim(tokens(columnseff(1))) /= 0 .and. len_trim(tokens(columnseff(2))) /= 0 ) then
125  read( tokens(columnseff(1)),* ) x(ndata)
126  read( tokens(columnseff(2)),* ) rbuffer
127 
128  y(ndata) = cmplx( rbuffer, 0.0_8 )
129 
130  ndata = ndata + 1
131  end if
132  end if
133  else
134  write(*,*) "### ERROR ### NFunction.fromFStream(). Bad number of tokens while reading file"
135  stop
136  end if
137  end if
138  end do
139 
140  npoints = ndata-1
141  call this.xgrid.fromarray( x(1:npoints) )
142  if( allocated(this.farray) ) deallocate(this.farray)
143  allocate( this.farray(npoints) )
144  this.farray = y(1:npoints)
145  call this.setunits( effunits )
146 
147  deallocate(x)
148  deallocate(y)
149  deallocate(columnseff)
150  end subroutine fromfstream

235  if( abs(this.farray( i )) > 1d-98 ) then

237  real(this.farray( i ))/effunits(2), &
238  aimag(this.farray( i ))/effunits(2)
239  else

241  end if
