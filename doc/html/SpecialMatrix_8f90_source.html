235  function specialmatrix_rotationtransform( axesBegin, axesEnd, angles ) result( R )
236  type(matrix), intent(in) :: axesbegin
237  type(matrix), intent(in) :: axesend
238  real(8), optional, intent(out) :: angles(3)
239  type(matrix) :: r
240 
241  type(matrix) :: u
242  real(8) :: rthetaphi(3)
243 
244  call u.columnvector( 3, values=axesbegin.data(:,3) )
245  u = u.projectionontonewaxes( axesend )
246  rthetaphi = math_cart2spher( u.data(:,1) )
247 
248  r = specialmatrix_rotation( rthetaphi(3), rthetaphi(2), 0.0_8 ) ! Ry(beta)*Rz(alpha)
249 
250  if( present(angles) ) then
251  angles(1) = rthetaphi(3) ! alpha
252  angles(2) = rthetaphi(2) ! beta
253  end if
254 
255  call u.columnvector( 3, values=axesbegin.data(:,1) )
256  u = u.projectionontonewaxes( axesend )
257 
258  u = r*u
259  rthetaphi = math_cart2spher( u.data(:,1) )
260 
261  r = specialmatrix_zrotation( rthetaphi(3) )*r ! Rz(gamma)*( Ry(beta)*Rz(alpha) )
262 
263  if( present(angles) ) then
264  angles(3) = rthetaphi(3) ! gamma
265  end if
