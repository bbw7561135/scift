84  subroutine run( this, rGrid, longRangeF, veryShortRangeF )
85  class(npotentialenergycurve) :: this
86  type(grid), intent(in) :: rgrid
87  procedure(prototypefunction), optional :: longrangef
88  procedure(prototypefunction), optional :: veryshortrangef
89 
90  type(spline) :: spline
91  real(8) :: xi
92  real(8), allocatable :: y(:)
93  integer :: i
94 
95  call spline.init( this.rawdata )
96  allocate( y(rgrid.npoints) )
97 
98  firstrawdata = [ this.rawdata.xgrid.data(1), this.rawdata.farray(1) ]
99  lastrawdata = [ this.rawdata.xgrid.data(this.rawdata.npoints()), this.rawdata.farray(this.rawdata.npoints()) ]
100 
101  do i=1,rgrid.npoints
102 
103  xi = rgrid.data(i)
104 
105  if( xi < firstrawdata(1) ) then
106  if( present(veryshortrangef) ) then
107  y(i) = veryshortrangef( xi )
108  else
109  y(i) = veryshortrangedefault( xi )
110  end if
111  else if( xi >= firstrawdata(1) .and. xi<= lastrawdata(1) ) then
112  y(i) = spline.evaluate(xi)
113  else if( xi > lastrawdata(1) ) then
114  if( present(longrangef) ) then
115  y(i) = longrangef( xi )
116  else
117  write(*,*) "@@ WARNING @@ Long range default function is not right implemented"
118  y(i) = longrangedefault( xi )
119  end if
120  end if
121 
122  end do
123 
124  call this.fromgridarray( rgrid, y )
125  end subroutine run

130  function shortrangedefault( x ) result( output )
131  real(8), intent(in) :: x
132  real(8) :: output
133 
134  real(8) :: re, we, wexe, rmass
135  real(8) :: de, alpha
136 
137  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
138  !! Cl2 X¹Σg+
139  !! NIST Standard Reference Data Program
140  !! -------------------------------------
141  !! re = 1.9879 Å
142  !! ωe = 559.72 cm⁻¹
143  !! ωexe = 2.675 cm⁻¹
144  !! m(Cl) = 35.4257 amu
145  !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
146  re = 1.9879_8*angs
147  we = 559.72_8*cm1
148  wexe = 2.675_8*cm1
149  rmass = 0.5_8*35.4257_8*amu
150 
151  de = 0.25_8*we**2.0_8/wexe
152  alpha = we*sqrt(0.5*rmass/de)
153 
154  output = de*( exp(2.0_8*alpha*(re-x)) &
155  - 2.0_8*exp(alpha*(re-x)) )
156  end function shortrangedefault
