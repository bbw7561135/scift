157

162  end if
163 
164  do i=1,command_argument_count()
165  call get_command_argument( i, buffer )
166 
167  if( buffer(1:1) == '-' .and. trim(buffer) == trim(param) .and. i+1<=command_argument_count() ) then
168  call get_command_argument( i+1, buffer )
169  output = trim(buffer)

171  end if
172  end do
173 
174  if( effdef /= "$$EMPTY**$$..@" ) then
175  output = trim(effdef)
