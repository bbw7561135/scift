252  subroutine build( this )
253  class(stringhistogram) :: this
254 
255  class(stringlistiterator), pointer :: iter
256  integer :: cvalue
257 
258  call this.counts.clear()
259  call this.density.clear()
260 
261  iter => this.begin
262  do while( associated(iter) )
263  cvalue = this.counts.at( iter.data, defaultvalue=0 )
264

266  call this.density.set( iter.data, real(cvalue+1,8)/real(this.size(),8) )
