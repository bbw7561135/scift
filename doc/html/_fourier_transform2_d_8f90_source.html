146  subroutine init( this, iFunc, domain, oFunc )
147  class(fouriertransform2d) :: this
148  class(cnfunction2d), target, intent(in) :: ifunc
149  integer(8), intent(in), optional :: domain
150  class(cnfunction2d), target, optional, intent(in) :: ofunc
151 
152  integer(8) :: effdomain
153 
154  integer :: nx, ny
155 
156  if( associated(this.ifunc) ) nullify(this.ifunc)
157  this.ifunc => ifunc
158  this.npoints = ifunc.npoints() ! << Hay que cambiar para hacer la transformada de algunos puntos
159 
160  nx = this.npoints(1)
161  ny = this.npoints(2)
162 
163  effdomain = fouriertransform_spatial_domain
164  if( present(domain) ) effdomain = domain
165 
166  if( present(ofunc) ) call goptions_error( "oFunc parameter is not implamented yet", "FourierTransform2D.init()" )
167 
168  call dfftw_plan_dft_2d( this.planf, nx, ny, this.ifunc.farray, this.ifunc.farray, fftw_forward, fftw_estimate )
169  call dfftw_plan_dft_2d( this.planb, nx, ny, this.ifunc.farray, this.ifunc.farray, fftw_backward, fftw_estimate )
170 
171  if( effdomain == fouriertransform_spatial_domain ) then
172 
173  this.x = this.ifunc.xygrid ! << Hay que cambiar para hacer la transformada de algunos puntos
174  this.omega = fouriertransform2d_omegagrid( this.x )
175 
176  else if( effdomain == fouriertransform_frequency_domain ) then
177 
178  call goptions_warning( "FourierTransform_FREQUENCY_DOMAIN have not been tested yet", "FourierTransform.init()" )
179  this.omega = this.ifunc.xygrid ! << Hay que cambiar para hacer la transformada de algunos puntos
180  this.x = fouriertransform2d_xygrid( this.omega )
181 
182  end if
183  end subroutine init

259  subroutine execute( this, sgn, sync, shift )
260  class(fouriertransform2d) :: this
261  integer, intent(in) :: sgn
262  logical, optional, intent(in) :: sync
263  logical, optional, intent(in) :: shift
264 
265  logical :: effsync
266  logical :: effshift
267 
268  effsync = .false.
269  if( present(sync) ) effsync = sync
270 
271  effshift = .false.
272  if( present(shift) ) effshift = shift
273 
274  if( sgn == fouriertransform_forward ) then
275 
276  call dfftw_execute( this.planf )
277 
278  if( effsync ) then
279  this.ifunc.xygrid = this.omega
280 
281  if( effshift ) then
282  call fouriertransform2d_phase( this.ifunc )
283  call fouriertransform2d_shift( this.ifunc )
284  end if
285  end if
286 
287  else if ( sgn == fouriertransform_backward ) then
288 
289  if( effsync ) then
290  if( effshift ) then
291  call fouriertransform2d_ishift( this.ifunc )
292  call fouriertransform2d_phase( this.ifunc )
293  end if
294  end if
295 
296  call dfftw_execute( this.planb )
297 
298  if( effsync ) then
299  this.ifunc.xygrid = this.x
300  end if
301 
302  this.ifunc = this.ifunc/real(this.nPoints(1)*this.nPoints(2),8)
303 
304  else
305  call goptions_error( "Bad value for sgn", "FourierTransform2D.execute()" )
306  end if
307  end subroutine execute
