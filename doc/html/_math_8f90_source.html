629  function math_erfinv( x_in ) result( r )
630  real(8), intent(in) :: x_in
631  real(8) :: r
632 
633  real(8) :: x, x2, y
634  integer :: sign_x
635 
636  x = x_in
637 
638  if( x < -1.0_8 .or. x > 1_8 ) then
639 ! r = Math_NAN
640  r = sqrt(-1.0_8)
641  return
642  end if
643 
644  if( x == 0.0_8 ) then
645  r = 0.0_8
646  return
647  end if
648 
649  if( x > 0.0_8 ) then
650  sign_x = 1
651  else
652  sign_x = -1
653  x = -x
654  end if
655 
656  if( x <= 0.7_8 ) then
657  x2 = x*x
658  r = x*(((erfinv_a3 * x2 + erfinv_a2) * x2 + erfinv_a1) * x2 + erfinv_a0)

668
