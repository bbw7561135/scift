29 !! In most of the textbooks, this form of the
30 !! Fourier transform is written as
31 !! F(u) = |F(u)|exp(i\theta(u)) (5)
32 !! but it’s essentially the same thing.
33 !!
34 !! There are some words that we use frequently when talking about Fourier
35 !! transform. The magnitude |F(u)|from equation 5 is called the Fourier spectrum
36 !! of f(x) and \theta(u) is phase angle. The square of the spectrum,
37 !! |F(u)|^2 = R(u)^2+I(u)^2 is often denoted as P(u) and is called the
38 !! power spectrum of f(x).
39 !!
40 !! The term spectral density is also commonly used to denote the power spectrum.
41 !! The Fourier spectrum is often plotted against values of ❩
42 !! . The Fourier spectrum
43 !! is useful because it can be easily plotted against ❩
44 !! on a piece of paper. (See figure
45 !! 4 for an example of the Fourier spectrum.) Note that XXXX ’s themselves are hard
46 !! to plot against ❩
47 !! on the 2-D plane because they are complex numbers

358  logical, optional, intent(in) :: shift

362 
363  effsync = .false.

366  effshift = .false.

374  this.ifunc.xgrid = this.omega

377  call fouriertransform_phase( this.ifunc )
378  call fouriertransform_shift( this.ifunc )
379  end if
380  end if

383 
384  if( effsync ) then
385  if( effshift ) then
386  call fouriertransform_ishift( this.ifunc )
387  call fouriertransform_phase( this.ifunc )
388  end if
389  end if
390 
391  call dfftw_execute( this.planb )
392 
393  if( effsync ) then
394  this.ifunc.xgrid = this.x
395  end if
396 
397  this.ifunc = this.ifunc/real(this.npoints,8)
398 
399  else
400  call goptions_error( "Bad value for sgn", "FourierTransform.execute()" )
401  end if
