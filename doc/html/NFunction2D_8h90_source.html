162  subroutine fromarrays( this, xArray, yArray, fArray, units, cutoff, fillWithZeros )
163  class(nfunction2d) :: this
164  real(8), intent(in) :: xarray(:)
165  real(8), intent(in) :: yarray(:)
166  __type_value__, intent(in) :: farray(:,:)
167  real(8), optional, intent(in) :: units(2)
168  real(8), optional, intent(in) :: cutoff
169  logical, optional :: fillwithzeros
170 
171  integer :: i, j
172  real(8) :: effunits(2)
173  real(8) :: effcutoff
174  logical :: efffillwithzeros
175 
176  effunits = [1.0_8, 1.0_8]
177  if( present(units) ) effunits = units
178 
179  effcutoff = 1d10
180  if( present(cutoff) ) effcutoff = cutoff
181 
182  efffillwithzeros = .false.
183  if( present(fillwithzeros) ) efffillwithzeros = fillwithzeros
184 
185  if( .not. efffillwithzeros ) then
186  if( size(xarray) /= size(farray,dim=1) ) then
187  call goptions_error( "The xArray's size should be equal to fArray's size", "NFunction2D.fromArrays()" )
188  end if
189 
190  if( size(yarray) /= size(farray,dim=2) ) then
191  call goptions_error( "The yArray's size should be equal to fArray's size", "NFunction2D.fromArrays()" )
192  end if
193  end if
194 
195  call this.xygrid.fromarray( xarray, yarray )
196  call this.xygrid.setunits( effunits(1) )
197 
198  if( allocated(this.farray) ) deallocate(this.farray)
199  allocate( this.farray( size(xarray), size(yarray) ) )
200 
201  do j=1,size(farray,dim=2); do i=1,size(farray,dim=1)
202 #if __ID_TYPE__ == 0
203  this.farray(i,j) = min( max( farray(i,j)*effunits(2), -effcutoff ), effcutoff )
204 #elif __ID_TYPE__ == 1
205  this.farray(i,j) = farray(i,j)*effunits(2)
206 #endif
207  end do; end do
208 
209  if( efffillwithzeros ) then
210  do j=size(farray,dim=2)+1,size(yarray); do i=size(farray,dim=1)+1,size(xarray)
211  this.farray(i,j) = 0.0_8
212  end do; end do
213  end if
214 
215  end subroutine fromarrays
