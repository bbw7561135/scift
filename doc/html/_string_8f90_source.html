426  subroutine fstring_split( str, tokens, delimiters )
427  character(*), intent(in) :: str
428  character(*), allocatable, intent(out) :: tokens(:)
429  character(*), intent(in) :: delimiters
430 
431  integer, allocatable :: postokenbegin(:), postokenend(:)
432  logical :: isdelimiter, isdelimiterprev
433  integer :: ntokens
434  integer :: i, j
435 
436  if( allocated(tokens) ) then
437  deallocate(tokens)
438  end if
439 
440  if( len(trim(str)) == 0 ) then
441  allocate( tokens(1) )
442  tokens(1) = str
443  return
444  end if
445 
446  ! En el peor de los casos todos los caracteres son separadores
447  allocate( postokenbegin(len(str)) )
448  allocate( postokenend(len(str)) )
449 
450  postokenbegin = 0
451  postokenend = 0
452 
453  !! scan
454  ntokens = 1
455  isdelimiterprev = .true.
456  do i=1,len(str)
457 ! write(*,"(3A)",advance="no") str(i:i)

464  end if
465  end do
466 
467  if( isdelimiter .and. .not. isdelimiterprev ) then

477 
478  isdelimiterprev = isdelimiter
479  end do

482  postokenend( ntokens ) = len( str )
